## Quick Example

```tsx
import { Component, Prop, Event } from '@pencel/runtime';

@Component({
  tag: 'ui-button'
})
export class UIButton extends HTMLElement {
  @Prop() label: string = 'Click me';

  @Event()
  declare clicked: CustomEvent<void>;

  render() {
    return (
      <button part="button" onClick={() => this.clicked.emit()}>
        {this.label}
      </button>
    );
  }
}
```
